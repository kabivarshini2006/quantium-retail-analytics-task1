# Cell 7: Extract & clean brand name
def extract_brand(name):
    # Take first word, fix known inconsistencies
    brand = name.split()[0].upper()
    corrections = {
        'RED': 'RRD', 'DORITO': 'DORITOS', 'SNBTS': 'SUNBITES',
        'INFZNS': 'INFUZIONS', 'WW': 'WOOLWORTHS', 'SMITH': 'SMITHS',
        'NCC': 'NATURAL', 'GRAIN': 'GRNWVES'
    }
    return corrections.get(brand, brand)

transaction['BRAND'] = transaction['PROD_NAME'].apply(extract_brand)

# Cell 12: Statistical test – Mainstream young/mid singles/couples pay more?
from scipy.stats import ttest_ind

mainstream = data[
    data['LIFESTAGE'].isin(['YOUNG SINGLES/COUPLES', 'MIDAGE SINGLES/COUPLES']) &
    (data['PREMIUM_CUSTOMER'] == 'Mainstream')
]
premium_budget = data[
    data['LIFESTAGE'].isin(['YOUNG SINGLES/COUPLES', 'MIDAGE SINGLES/COUPLES']) &
    (data['PREMIUM_CUSTOMER'] != 'Mainstream')
]

main_price = mainstream['TOT_SALES'] / mainstream['PROD_QTY']
other_price = premium_budget['TOT_SALES'] / premium_budget['PROD_QTY']

t_stat, p_val = ttest_ind(main_price, other_price, equal_var=False)
print(f"t-test p-value: {p_val:.2e} → significantly higher price in mainstream")

# Cell 13: Deep dive – affinity for young mainstream singles/couples
target = data[(data['LIFESTAGE'] == 'YOUNG SINGLES/COUPLES') & (data['PREMIUM_CUSTOMER'] == 'Mainstream')]
other  = data[~((data['LIFESTAGE'] == 'YOUNG SINGLES/COUPLES') & (data['PREMIUM_CUSTOMER'] == 'Mainstream'))]

brand_aff = (
    target.groupby('BRAND')['PROD_QTY'].sum() / target['PROD_QTY'].sum()
).to_frame('target_share').join(
    other.groupby('BRAND')['PROD_QTY'].sum() / other['PROD_QTY'].sum()
).rename(columns={'PROD_QTY':'other_share'})
brand_aff['affinity'] = brand_aff['target_share'] / brand_aff['other_share']
print(brand_aff.sort_values('affinity', ascending=False).head(10))

# Same for pack size
pack_aff = (
    target.groupby('PACK_SIZE')['PROD_QTY'].sum() / target['PROD_QTY'].sum()
).to_frame('target_share').join(
    other.groupby('PACK_SIZE')['PROD_QTY'].sum() / other['PROD_QTY'].sum()
).rename(columns={'PROD_QTY':'other_share'})
pack_aff['affinity'] = pack_aff['target_share'] / pack_aff['other_share']
print(pack_aff.sort_values('affinity', ascending=False).head(8))
